rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Orders collection - simple rules for counter sales
    match /orders/{orderId} {
      allow read, write: if true; // Allow anyone to read/write orders (for counter sales)
    }

    // Expenses collection - only authenticated users
    match /expenses/{expenseId} {
      allow read, write: if request.auth != null;
    }

    // Settings collection - only authenticated users
    match /settings/{settingId} {
      allow read, write: if request.auth != null;
    }

    // Daily summaries collection - only authenticated users
    match /daily_summaries/{summaryId} {
      allow read, write: if request.auth != null;
    }

    // Menu items collection - read for everyone, write for authenticated users
    match /menu_items/{itemId} {
      allow read: if true; // Allow anyone to read menu items (for customer orders)
      allow write: if request.auth != null; // Only authenticated users can manage menu items
    }

    // Users collection - allow registration process
    match /users/{userId} {
      allow read: if true; // Allow read for email checking
      allow create: if false; // Prevent new document creation
      allow update: if request.auth != null; // Allow authenticated users to update
    }
  }
}
